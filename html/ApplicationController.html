    current_user() click to toggle source

    This function allows to access to the current user which is authenticate on the website. If there is no user which is authenticate, this function return nil. You have to use this function to check the authentification of a user.

def current_user
  begin
    @current_user ||= User.find_by_auth_token!(cookies[:auth_token]) if cookies[:auth_token]
  rescue ActiveRecord::RecordNotFound => e
    @current_user ||= nil
  end
end

def get_auth_token_user_id()
  if request.headers["Authorization"]
    @token = request.headers["Authorization"].scan(/\"(.*?)\"/)
    @user = User.where(:auth_token => @token).first()
  end
  return @user ? @user.id : session[:user_id]
end
