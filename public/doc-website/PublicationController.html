def index
        param = Hash.new
        param["publication_id"] = params[:id]
        response = WS.get("/comments.json", param, cookies[:auth_token])
        @comments = JSON.parse(response.body)["result"]["comments"]

        url = "/publications/" + params[:id].to_s + ".json"
        response = WS.get(url, nil)
        @publication = JSON.parse(response.body)["result"]["publication"]

        url = "/places/" + @publication["place"]["id"] + ".json"
        response = WS.get(url, nil)          
        @place = JSON.parse(response.body)["result"]["place"]
end

def new
        @param = params
        param = Hash.new
        param["publication[place_id]"] = params[:place_id]
        param["publication[user_latitude]"] = params[:latitude]
        param["publication[user_longitude]"] = params[:longitude]
        param["publication[content]"] = params[:content]
        param["publication[file]"] = params[:file]
        response = WS.post("/publications.json", param, cookies[:auth_token], true)
        if response.nil? == false
                body = JSON.parse(response.body)
        end
        if body["responseCode"] == 0
                redirect_to "/place/" + params["place_id"]
        else
                redirect_to "/place/" + params["place_id"], :flash => { :errors => body["responseMessage"] }
        end
end
