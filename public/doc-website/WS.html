def delete(route, auth = nil, params)
        begin
                uri = create_url(route)
                if params.nil? == false
                        uri.query = URI.encode_www_form(params)
                end
                http = Net::HTTP.new(uri.host, uri.port)
                request = Net::HTTP::Delete.new(uri.request_uri)
                if (auth.nil? == false)
                        header_value = "Token token=\"" + auth + "\""
                        request.add_field("Authorization", header_value)                           
                end
                response = http.request(request)
        rescue
                nil
        end
end

def get_errors_from_body(body)
        errors = Array.new 
        # errors.push(body["responseMessage"])
        body["result"]["error"].each { |key, value| errors.push(key + ": " + body["result"]["error"][key][0]) }
        errors
end

    This method send a post request at /neerbyy/. The params parameter contains all the parameters needed in the request The auth parameter is either nil or the auth_token. The multipart parameter is a boolean to know if the request is a multipart or not (Multipart means that you can send a file)

    # File lib/WS.rb, line 15
def post(route, params, auth = nil, multipart = false)
        # uri = create_url(route)
        # response = Net::HTTP.post_form(uri, params)
        begin
                uri = create_url(route)
                uri.query = URI.encode_www_form(params)
                http = Net::HTTP.new(uri.host, uri.port)
                request = Net::HTTP::Post.new(uri.request_uri)
                if (multipart == true)
                        request.content_type = 'multipart/form-data'
                end
                if (auth.nil? == false)
                        header_value = "Token token=\"" + auth + "\""
                        request.add_field("Authorization", header_value)   
                end                 
                response = http.request(request)
        rescue Exception => msg
                msg
        end
end
