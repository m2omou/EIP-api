def new
        param = Hash.new
        param["email"] = params[:email]
        response = WS.post("/password_resets.json", param, nil)
        body = JSON.parse(response.body)
        if (body["responseCode"] == 0)
                redirect_to "/log_in", :flash => { :success => "An email has been sent to your email adress. Follow the instruction in order to reset your password."}
        else
                redirect_to "/log_in", :flash => { :errors => "The email " + param["email"] + " does not exist in our database." }
        end
end
